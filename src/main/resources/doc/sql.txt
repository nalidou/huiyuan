
DROP DATABASE `huiyuan`;
CREATE DATABASE huiyuan CHARSET utf8;

USE huiyuan;

DROP TABLE member;
CREATE TABLE member(
id BIGINT(20) NOT NULL AUTO_INCREMENT,
NAME VARCHAR(255) NOT NULL,
sex INT(10),
age INT(10),
nation VARCHAR(50),
eduBackground VARCHAR(50),
nativeAddress VARCHAR(50),
birthday VARCHAR(50),
idCard VARCHAR(255),
mobilephone VARCHAR(50),
telephone VARCHAR(50),
email VARCHAR(50),
homeAdderss VARCHAR(255),

companyName VARCHAR(255),
companyJob VARCHAR(255),
companyAddress VARCHAR(255),
companyStatus INT(10),

memberName VARCHAR(255),
memberJob VARCHAR(50),
memberProperty VARCHAR(50),
memberEnterDate BIGINT(20),
memberReferrer VARCHAR(50),
memberStatus INT(10),

isQuit INT(10),
memo TEXT,
operaRecord VARCHAR(50),

createTime BIGINT(20),
userId BIGINT(20),
info TEXT,
isDelete INT(10),

INDEX(NAME),
INDEX(mobilephone),
INDEX(companyJob),
INDEX(memberProperty),
INDEX(memberEnterDate),
PRIMARY KEY(id)
)CHARSET utf8;


CREATE TABLE member_job (
id BIGINT(20) NOT NULL AUTO_INCREMENT,
NAME VARCHAR(255),

info TEXT,
userId BIGINT(20),
createTime BIGINT(20),

PRIMARY KEY(id),
INDEX(NAME),
INDEX(userId)
)CHARSET utf8;


CREATE TABLE member_property (
id BIGINT(20) NOT NULL AUTO_INCREMENT,
NAME VARCHAR(255),

createTime BIGINT(20),
info TEXT,
userId BIGINT(20),

PRIMARY KEY(id),
INDEX(NAME),
INDEX(userId)
)CHARSET utf8;



CREATE TABLE USER (
id BIGINT(20) NOT NULL AUTO_INCREMENT ,
account VARCHAR(255) NOT NULL,
PASSWORD VARCHAR(255) NOT NULL,
LEVEL INT(10) NOT NULL,
isUsed INT(10) NOT NULL,

createTime BIGINT(20) NOT NULL,
info TEXT,

PRIMARY KEY(id),
UNIQUE KEY(account)
)CHARSET utf8;
